CREATE TABLE patients (
 id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 first_name VARCHAR(64),
 last_name VARCHAR(64),
 blood_type VARCHAR(3),
 address VARCHAR(1024),
 dob DATETIME
);

DESCRIBE patients;

CREATE TABLE diagnosis (
 id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 description TEXT,
 date DATETIME
);
